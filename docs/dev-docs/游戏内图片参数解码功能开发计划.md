# 游戏内图片参数解码功能开发计划

> **For agentic workers:** 本计划包含 10 个任务，建议按顺序执行。每个任务产出可独立验证。

**Goal:** 为无限暖暖相册管理工具增加游戏内图片参数解码功能，支持在图片详情页和全屏查看器中展示相机参数、拍摄地点等信息，并提供"复制相机参数"能力。

**Architecture:** 主进程通过 `koffi` 加载 `nuan5_decryption.dll` 完成 FFI 调用，渲染进程通过 IPC 异步获取解密结果。解密后的数据在主进程完成解析和转换，以简化 JSON 形式返回给渲染进程展示。采用缓存机制避免重复解密同一文件。

**Tech Stack:** koffi (FFI), Electron IPC, React + TypeScript, JSON.parse (数据解析)

- [ ] **Step 2: 创建解密服务**

`src/main/services/decryption-service.ts`:

```typescript
import koffi from 'koffi'
import path from 'path'
import fs from 'fs'
import { app } from 'electron'
import { logger } from '../utils/logger'
import {
  DecryptionStatus,
  MediaParamType,
  ALBUM_TYPE_MAP,
  type DecryptionResult,
  type CameraParams,
} from '../types/decryption'
import { resolveNativePath } from '../utils/native-path'

// ============================================================
// DLL 加载与类型定义
// ============================================================

/** 定位 DLL 路径（优先 asar.unpacked，fallback 到 resources） */
function findDllPath(): string {
  const candidates = [
    // asar.unpacked 目录（打包后）
    path.join(process.resourcesPath, 'nuan5_decryption.dll'),
    // 开发环境
    path.join(app.getAppPath(), 'resources', 'nuan5_decryption.dll'),
  ]
  for (const p of candidates) {
    if (fs.existsSync(p)) return p
  }
  throw new Error('nuan5_decryption.dll not found')
}

let lib: koffi.IKoffiLib | null = null
let abiVersionFunc: koffi.KoffiFunction | null = null

/** 初始化 DLL（惰性加载） */
function getLib(): koffi.IKoffiLib {
  if (lib) return lib
  const dllPath = findDllPath()
  logger.info('[decryption] Loading DLL:', dllPath)
  lib = koffi.load(dllPath)

  // 验证 ABI 版本
  const abiVer = (lib.func('abi_version', 'uint32_t', []) as koffi.KoffiFunction)()
  logger.info('[decryption] DLL ABI version:', abiVer)

  return lib
}

// C 结构体定义
const CBytes = koffi.struct('CBytes', {
  data: koffi.pointer('uint8_t'),
  len: 'size_t',
  cap: 'size_t',
})

const MediaDecryptionResult = koffi.struct('MediaDecryptionResult', {
  status: 'uint32_t',
  bytes: CBytes,
})

const MediaKeyPtr = koffi.pointer('void')

// ============================================================
// 相机参数解析（CameraParams V1/V2）
// ============================================================

/**
 * 解析解密后的原始字节为相机参数。
 * 根据 PoC 验证，CameraParams 解密后是类似 JSON 数组的文本格式。
 * 例如: [1,0,21,128.61,151.33,5.26,-99.27,...]
 */
function parseCameraParams(data: Buffer): CameraParams | undefined {
  try {
    const text = data.toString('utf-8').trim()
    // 尝试 JSON 解析
    const arr = JSON.parse(text) as unknown[]
    if (!Array.isArray(arr) || arr.length < 31) {
      logger.warn('[decryption] CameraParams array too short:', arr.length)
      return undefined
    }

    // V1 格式索引映射（见 nikki_albums converter.rs）
    const rawParams = text

    return {
      focalLength: Number(arr[14]) || 0,
      apertureSection: Number(arr[15]) || 0,
      brightness: Number(arr[22]) || 0,
      exposure: Number(arr[23]) || 0,
      contrast: Number(arr[24]) || 0,
      saturation: Number(arr[25]) || 0,
      vibrance: Number(arr[26]) || 0,
      highlights: Number(arr[27]) || 0,
      shadows: Number(arr[28]) || 0,
      vignetteIntensity: Number(arr[19]) || 0,
      bloomIntensity: Number(arr[20]) || 0,
      bloomThreshold: Number(arr[21]) || 0,
      portraitMode: Number(arr[1]) === 1,
      filter: arr[29] && arr[29] !== 'None'
        ? { id: String(arr[29]), strength: Number(arr[30]) || 0 }
        : undefined,
      light: arr[17] && arr[17] !== 'None'
        ? { id: String(arr[17]), strength: Number(arr[18]) || 0 }
        : undefined,
      rawParams,
    }
  } catch (err) {
    logger.error('[decryption] Failed to parse camera params:', err)
    return undefined
  }
}

// ============================================================
// 对外接口
// ============================================================

/**
 * 从文件路径解密参数。
 * @param filePath 游戏截图文件路径
 * @param albumType 相册类型（决定 Flag 和密钥类型）
 * @param uid 玩家 UID（照片类参数需要，CameraParams 不需要）
 */
export function decodeFileParams(
  filePath: string,
  albumType: string,
  uid?: string
): DecryptionResult {
  try {
    const libInstance = getLib()

    // 绑定函数（每次调用都重新绑定，避免 koffi 的指针生命周期问题）
    const mediaKeyCameraParam = libInstance.func('media_key_camera_param', MediaKeyPtr, [])
    const mediaKeyFromStr = libInstance.func('media_key_from_str', MediaKeyPtr, [koffi.pointer('char')])
    const freeMediaKey = libInstance.func('free_media_key', 'void', [MediaKeyPtr])
    const mediaDecodeFileUnchecked = libInstance.func('media_decode_file_unchecked', MediaDecryptionResult, [
      koffi.pointer('uint8_t'),
      'size_t',
      koffi.pointer('char'),
      MediaKeyPtr,
    ])
    const freeCBytes = libInstance.func('free_c_bytes', 'void', [CBytes])

    // 确定参数类型和 Flag
    const paramType = ALBUM_TYPE_MAP[albumType] || MediaParamType.NikkiPhoto
    const flag = paramType === MediaParamType.CameraParams
      ? Buffer.alloc(0)
      : Buffer.from([0xFF, 0xD9]) // JPEG 结束标记

    // 创建密钥
    let key: koffi.PointerObject | null = null
    if (paramType === MediaParamType.CameraParams) {
      key = mediaKeyCameraParam() as koffi.PointerObject
    } else if (uid) {
      const cStr = Buffer.from(uid + '\0', 'utf-8')
      key = (mediaKeyFromStr as koffi.KoffiFunction)(cStr) as koffi.PointerObject
    } else {
      return { hasParams: false, error: 'UID 未提供，无法解密照片参数' }
    }

    if (!key) {
      return { hasParams: false, error: '密钥创建失败' }
    }

    try {
      // 调用 DLL 解密
      const result = (mediaDecodeFileUnchecked as koffi.KoffiFunction)(
        flag,
        flag.length,
        Buffer.from(filePath + '\0', 'utf-8'),
        key
      ) as { status: number; bytes: { data: koffi.PointerObject; len: number; cap: number } }

      if (result.status !== DecryptionStatus.Success) {
        const statusMap: Record<number, string> = {
          [DecryptionStatus.DataLenIsNotAMultipleOf16]: '数据长度不是 16 的倍数',
          [DecryptionStatus.DecodingBase64Failed]: 'Base64 解码失败',
          [DecryptionStatus.FindNoStartFlag]: '未找到开始标志',
          [DecryptionStatus.FindNoEndFlag]: '未找到结束标志',
          [DecryptionStatus.Io]: '文件读取失败',
          [DecryptionStatus.IllegalUTF8]: '非法 UTF-8',
        }
        return {
          hasParams: false,
          error: statusMap[result.status] || `解密失败 (status=${result.status})`,
        }
      }

      // 提取解密后的数据
      if (!result.bytes.data || result.bytes.len === 0) {
        return { hasParams: false }
      }

      // 从指针读取字节数据
      let decryptedBuf: Buffer
      try {
        // koffi.decode 读取指针指向的内存
        const rawBytes = koffi.decode(result.bytes.data, 'uint8_t', result.bytes.len) as Uint8Array
        decryptedBuf = Buffer.from(rawBytes)
      } catch {
        // 兜底：直接读取（某些 koffi 版本行为不同）
        decryptedBuf = Buffer.alloc(result.bytes.len)
        // 此处无法安全读取，依赖 koffi.decode
        return { hasParams: false, error: '无法读取解密数据' }
      }

      // 释放 DLL 分配的内存
      try {
        freeCBytes(result.bytes)
      } catch (e) {
        logger.warn('[decryption] free_c_bytes failed:', e)
      }

      // 解析参数
      const camera = parseCameraParams(decryptedBuf)

      return {
        hasParams: !!camera,
        camera,
      }
    } finally {
      // 释放密钥
      if (key) {
        try { freeMediaKey(key) } catch { /* ignore */ }
      }
    }
  } catch (err) {
    logger.error('[decryption] Unexpected error:', err)
    return { hasParams: false, error: err instanceof Error ? err.message : '未知错误' }
  }
}

/** 关闭解密服务，释放 DLL */
export function disposeDecryptionService(): void {
  if (lib) {
    try {
      lib.unload()
    } catch { /* ignore */ }
    lib = null
  }
}
```

import { useState, useEffect, useRef, useCallback } from 'react'
import type { GameParamsData } from '../types/decryption'

interface UseGameParamsResult {
  data: GameParamsData | null
  loading: boolean
  error: string | null
  refresh: () => void
}

/** 内存缓存：以 filePath 为键 */
const cache = new Map<string, GameParamsData>()

export function useGameParams(
  filePath: string | undefined,
  albumType: string | undefined,
  uid: string | undefined,
  enabled: boolean = true
): UseGameParamsResult {
  const [data, setData] = useState<GameParamsData | null>(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const tokenRef = useRef({ cancelled: false })

  const doFetch = useCallback(() => {
    if (!filePath || !albumType || !enabled) {
      setData(null)
      setError(null)
      return
    }

    // 检查缓存
    const cacheKey = `${filePath}::${uid || ''}`
    if (cache.has(cacheKey)) {
      setData(cache.get(cacheKey)!)
      setLoading(false)
      setError(null)
      return
    }

    const token = { cancelled: false }
    tokenRef.current = token
    setLoading(true)
    setError(null)

    window.electronAPI.decrypt
      .decodeFile(filePath, albumType, uid)
      .then((result) => {
        if (token.cancelled) return
        if (result.success && result.data) {
          cache.set(cacheKey, result.data)
          setData(result.data)
        } else {
          setData({ hasParams: false, error: result.message || '解码失败' })
        }
      })
      .catch((err: unknown) => {
        if (token.cancelled) return
        setData({ hasParams: false, error: err instanceof Error ? err.message : '未知错误' })
      })
      .finally(() => {
        if (!token.cancelled) setLoading(false)
      })
  }, [filePath, albumType, uid, enabled])

  useEffect(() => {
    doFetch()
    return () => {
      tokenRef.current.cancelled = true
    }
  }, [doFetch])

  const refresh = useCallback(() => {
    if (filePath && albumType) {
      const cacheKey = `${filePath}::${uid || ''}`
      cache.delete(cacheKey)
    }
    doFetch()
  }, [filePath, albumType, uid, doFetch])

  return { data, loading, error, refresh }
}
```

import { useGameParams } from '../../hooks/useGameParams'
import { IconCopyText, IconInfo } from '../../icons'

interface GameParamsPanelProps {
  file: MediaFile
  variant?: 'light' | 'dark'
  showTitle?: boolean
}

interface ParamRow {
  label: string
  value: string
}

function cameraToRows(camera: NonNullable<ReturnType<typeof useGameParams>['data']>['camera']): ParamRow[] {
  if (!camera) return []
  const rows: ParamRow[] = []
  if (camera.focalLength !== undefined) rows.push({ label: '焦距', value: `${camera.focalLength.toFixed(1)}mm` })
  if (camera.apertureSection !== undefined) rows.push({ label: '光圈', value: `F${camera.apertureSection}` })
  if (camera.brightness !== undefined) rows.push({ label: '亮度', value: `${(camera.brightness * 100).toFixed(0)}%` })
  if (camera.exposure !== undefined) rows.push({ label: '曝光', value: `${(camera.exposure * 100).toFixed(0)}%` })
  if (camera.contrast !== undefined) rows.push({ label: '对比度', value: `${(camera.contrast * 100).toFixed(0)}%` })
  if (camera.saturation !== undefined) rows.push({ label: '饱和度', value: `${(camera.saturation * 100).toFixed(0)}%` })
  if (camera.vibrance !== undefined) rows.push({ label: '鲜艳度', value: `${(camera.vibrance * 100).toFixed(0)}%` })
  if (camera.highlights !== undefined) rows.push({ label: '高光', value: `${(camera.highlights * 100).toFixed(0)}%` })
  if (camera.shadows !== undefined) rows.push({ label: '阴影', value: `${(camera.shadows * 100).toFixed(0)}%` })
  if (camera.vignetteIntensity !== undefined) rows.push({ label: '暗角', value: `${(camera.vignetteIntensity * 100).toFixed(0)}%` })
  if (camera.bloomIntensity !== undefined) rows.push({ label: '泛光强度', value: `${(camera.bloomIntensity * 100).toFixed(0)}%` })
  if (camera.bloomThreshold !== undefined) rows.push({ label: '泛光阈值', value: `${(camera.bloomThreshold * 100).toFixed(0)}%` })
  if (camera.portraitMode) rows.push({ label: '人像模式', value: '开启' })
  if (camera.filter) rows.push({ label: '滤镜', value: `${camera.filter.id} (${(camera.filter.strength * 100).toFixed(0)}%)` })
  if (camera.light) rows.push({ label: '打光', value: `${camera.light.id} (${(camera.light.strength * 100).toFixed(0)}%)` })
  return rows
}

export const GameParamsPanel: React.FC<GameParamsPanelProps> = ({ file, variant = 'light', showTitle = true }) => {
  // 仅对游戏内图片启用（file_type === 'image' 且 album_type 存在）
  const enabled = file.file_type === 'image' && !!file.album_type
  const { data, loading, error } = useGameParams(
    file.file_path,
    file.album_type,
    file.account_uid,
    enabled
  )
  const [copied, setCopied] = useState(false)

  const rows = useMemo(() => cameraToRows(data?.camera), [data?.camera])

  const isDark = variant === 'dark'
  const themeStyles = isDark
    ? {
        container: 'rounded-xl',
        title: 'text-white/90',
        row: 'border-white/10',
        label: 'text-white/55',
        value: 'text-white/95',
        empty: 'text-white/50',
        btn: 'text-white/80 hover:bg-white/10',
        copied: 'text-emerald-400'
      }
    : {
        container: '',
        title: '',
        row: 'border-black/5',
        label: '',
        value: '',
        empty: '',
        btn: '',
        copied: ''
      }

  const handleCopy = async () => {
    const raw = data?.camera?.rawParams
    if (!raw) return
    try {
      await navigator.clipboard.writeText(raw)
      setCopied(true)
      setTimeout(() => setCopied(false), 1500)
    } catch {
      const textarea = document.createElement('textarea')
      textarea.value = raw
      textarea.style.position = 'fixed'
      textarea.style.opacity = '0'
      document.body.appendChild(textarea)
      textarea.select()
      try {
        document.execCommand('copy')
        setCopied(true)
        setTimeout(() => setCopied(false), 1500)
      } catch { /* ignore */ }
      finally { document.body.removeChild(textarea) }
    }
  }

  if (!enabled) return null

  return (
    <div className={themeStyles.container}>
      {showTitle && (
        <div className="flex items-center justify-between mb-2">
          <div className="flex items-center gap-1.5">
            <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
              <path d="M23 19a2 2 0 0 1-2 2H3a2 2 0 0 1-2-2V8a2 2 0 0 1 2-2h4l2-3h6l2 3h4a2 2 0 0 1 2 2z" />
              <circle cx="12" cy="13" r="4" />
            </svg>
            <span className={`text-xs font-semibold ${themeStyles.title}`} style={!isDark ? { color: 'var(--text-secondary)' } : undefined}>
              游戏拍摄参数
            </span>
          </div>
          {rows.length > 0 && (
            <button
              className={`text-xs flex items-center gap-1 px-2 py-1 rounded transition-colors ${themeStyles.btn}`}
              onClick={handleCopy}
              title="复制完整相机参数"
              aria-label="复制相机参数"
            >
              {copied ? (
                <span className={themeStyles.copied}>已复制</span>
              ) : (
                <>
                  <IconCopyText size={12} />
                  <span>复制参数</span>
                </>
              )}
            </button>
          )}
        </div>
      )}

      {loading && (
        <div className={`text-xs py-3 flex items-center gap-2 ${themeStyles.empty}`} style={!isDark ? { color: 'var(--text-tertiary)' } : undefined}>
          <div className="w-3 h-3 rounded-full border border-current border-t-transparent animate-spin" />
          正在解析游戏参数...
        </div>
      )}

      {!loading && error && (
        <div className={`text-xs py-3 ${themeStyles.empty}`} style={!isDark ? { color: 'var(--text-tertiary)' } : undefined}>
          参数解析失败: {error}
        </div>
      )}

      {!loading && !error && rows.length === 0 && (
        <div className={`text-xs py-3 flex items-center gap-1.5 ${themeStyles.empty}`} style={!isDark ? { color: 'var(--text-tertiary)' } : undefined}>
          <IconInfo size={12} />
          此图片未包含游戏拍摄参数
        </div>
      )}

      {!loading && !error && rows.length > 0 && (
        <div className="space-y-1">
          {rows.map((row) => (
            <div
              key={row.label}
              className={`flex items-baseline gap-3 text-xs py-1 border-b last:border-b-0 ${themeStyles.row}`}
            >
              <span
                className={`flex-shrink-0 w-20 ${themeStyles.label}`}
                style={!isDark ? { color: 'var(--text-tertiary)' } : undefined}
              >
                {row.label}
              </span>
              <span
                className={`flex-1 break-all font-mono ${themeStyles.value}`}
                style={!isDark ? { color: 'var(--text-primary)' } : undefined}
              >
                {row.value}
              </span>
            </div>
          ))}
        </div>
      )}
    </div>
  )
}
```

渲染进程类型检查通过。
